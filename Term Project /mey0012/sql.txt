1. SELECT Subj.CategoryName FROM db_book Book, db_subject Subj, db_supplier Supp WHERE Book.SupplierID = Supp.SupplierID AND Book.SubjectID = Subj.SubjectID AND Supp.CompanyName = 'supplier2';
2. SELECT Book.Title, Book.UnitPrice From db_book Book, db_supplier Supp Where Book.SupplierID = Supp.SupplierID AND Book.UnitPrice = (select MAX(Book2.UnitPrice) From db_book Book2, db_supplier Supp2 Where Book2.SupplierID = Supp2.SupplierID AND Book2.UnitPrice AND Supp2.CompanyName = 'supplier3');
3. SELECT Distinct Book.Title FROM db_book Book, db_order Orde, db_order_detail OrderDet, db_customer Cust WHERE Book.BookID = OrderDet.BookID AND OrderDet.OrderID = Orde.OrderID AND Orde.CustomerID = Cust.CustomerID AND Cust.FirstName='firstname1' AND Cust.LastName='lastname1';
4. SELECT Book.Title FROM db_book Book WHERE Book.Quantity > 10;
5. SELECT SUM(OrderDet.Quantity * Book.UnitPrice) FROM db_book Book, db_order_detail OrderDet, db_order Orde, db_customer Cust WHERE OrderDet.BookID = Book.BookID AND Orde.OrderID = OrderDet.OrderID AND Cust.CustomerID = Orde.CustomerID AND Cust.FirstName = "firstname1" AND Cust.LastName='lastname1';
6. SELECT FirstName, LastName from (SELECT Cust.FirstName as FirstName, Cust.LastName as LastName, SUM(OrderDet.Quantity * Book.UnitPrice) AS Total FROM db_book Book, db_order_detail OrderDet, db_order Orde, db_customer Cust WHERE OrderDet.BookID = Book.BookID AND Orde.OrderID = OrderDet.OrderID AND Cust.CustomerID = Orde.CustomerID GROUP BY Cust.CustomerID HAVING Total < 80) as current;
7. SELECT Book.Title FROM db_book Book, db_supplier Supp WHERE Book.SupplierID = Supp.SupplierID AND Supp.CompanyName = 'supplier2';
8. SELECT Cust.FirstName, Cust.LastName, SUM(OrderDet.Quantity * Book.UnitPrice) AS Total FROM db_book Book, db_order_detail OrderDet, db_order Orde, db_customer Cust  WHERE OrderDet.BookID = Book.BookID AND Orde.OrderID = OrderDet.OrderID AND Cust.CustomerID = Orde.CustomerID GROUP BY Cust.CustomerID ORDER BY Total DESC;
9. SELECT Book.Title, Ship.ShpperName FROM db_book Book, db_shipper Ship, db_order Orde, db_order_detail OrderDet WHERE Book.BookID = OrderDet.BookID AND OrderDet.OrderID = Orde.OrderID AND Orde.ShipperID = Ship.ShipperID AND Orde.ShippedDate = "2016-08-04";
10. SELECT Distinct Book.Title FROM db_book Book, db_order Orde, db_order_detail OrderDet, db_customer Cust WHERE Book.BookID = OrderDet.BookID AND OrderDet.OrderID = Orde.OrderID AND Orde.CustomerID = Cust.CustomerID AND Cust.FirstName='firstname1' AND Cust.LastName='lastname1' AND Book.BookID in (SELECT Book1.BookID FROM db_book Book1, db_order Orde1, db_order_detail OrderDet1, db_customer Cust1 WHERE Book1.BookID = OrderDet1.BookID AND OrderDet1.OrderID = Orde1.OrderID AND Orde1.CustomerID = Cust1.CustomerID AND Cust1.FirstName='firstname4' AND Cust1.LastName='lastname4');
11. SELECT Book.Title FROM db_book Book, db_employee Empl, db_order Orde, db_order_detail OrderDet WHERE OrderDet.BookID = Book.BookID AND OrderDet.OrderID = Orde.OrderID AND Orde.EmployeeID = Empl.EmployeeID AND Empl.FirstName = 'firstname6' AND LastName='lastname6';
12. SELECT Book.Title, OrderDet.Quantity FROM db_book Book, db_order_detail OrderDet WHERE Book.BookID = OrderDet.BookID ORDER BY OrderDet.Quantity ASC;
13. SELECT FirstName, LastName from (select Cust.FirstName as FirstName, Cust.LastName as LastName, SUM(OrderDet.Quantity) AS Total FROM db_customer Cust, db_order Orde, db_order_detail OrderDet WHERE OrderDet.OrderID = Orde.orderID AND Orde.CustomerID = Cust.CustomerID GROUP BY Cust.CustomerID HAVING Total >= 2) as current;
14. SELECT Cust.FirstName, Cust.LastName, Book.Title FROM db_customer Cust, db_subject Subj, db_book Book, db_order Orde, db_order_detail OrderDet WHERE Orde.OrderID = OrderDet.OrderID AND Book.BookID = OrderDet.BookID AND Orde.CustomerID = Cust.CustomerID AND Book.SubjectID = Subj.SubjectID AND (Subj.CategoryName = 'category3' OR Subj.CategoryName = 'category4');
15. SELECT Cust.FirstName, Cust.LastName FROM db_order Orde, db_order_detail OrderDet, db_customer Cust, db_book Book WHERE Orde.OrderID = OrderDet.OrderID AND Orde.CustomerID = Cust.CustomerID AND OrderDet.BookID = Book.BookID AND Book.Author = 'author1';
16. SELECT Empl.FirstName, Empl.LastName, SUM(Book.UnitPrice * OrderDet.Quantity) AS Price FROM db_book Book, db_employee Empl, db_order Orde, db_order_detail OrderDet WHERE Orde.OrderID = OrderDet.OrderID AND Book.BookID = OrderDet.BookID AND Orde.EmployeeID = Empl.EmployeeID GROUP BY Empl.EmployeeID;
17. SELECT Book.Title, SUM(OrderDet.Quantity) AS Total FROM db_book Book, db_order Orde, db_order_detail OrderDet  WHERE Book.BookID = OrderDet.BookID AND Orde.OrderID = OrderDet.OrderID AND (Orde.ShippedDate IS NULL OR Orde.ShippedDate > '2016-08-04') GROUP BY Book.Title;
18. SELECT Cust.FirstName, Cust.LastName, SUM(OrderDet.Quantity) AS Total FROM db_customer Cust, db_order Orde, db_order_detail OrderDet WHERE OrderDet.OrderID = Orde.orderID AND Orde.CustomerID = Cust.CustomerID GROUP BY Cust.CustomerID HAVING Total > 1 ORDER BY Total DESC;
19. SELECT Cust.FirstName, Cust.LastName, Cust.Phone, SUM(OrderDet.Quantity) AS Total FROM db_customer Cust, db_order Orde, db_order_detail OrderDet WHERE OrderDet.OrderID = Orde.orderID AND Orde.CustomerID = Cust.CustomerID GROUP BY Cust.CustomerID HAVING Total > 3;
